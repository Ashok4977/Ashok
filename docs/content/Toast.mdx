```javascript live noinline
const DemoComponent = () => {
  const toasts = useToasts({autoDismiss: true});
  const states = ['success', 'default', 'warning', 'error', 'loading']
  const messages = ['hello!', 'oopsies', 'try again plz', 'nice job!']

  const handleAddToastClick = () => {
    toasts.addToast({
      message: messages[Math.floor(Math.random() * Math.floor(4))],
      type: states[Math.floor(Math.random() * Math.floor(5))]
    })
  }

  return (
    <>
      <Button onClick={handleAddToastClick}>Add a toast</Button>
      <Absolute bottom={4} left={4}>
        <ToastContainer {...toasts}/>
      </Absolute>
    </>
  )
}

render(<DemoComponent/>)
```
