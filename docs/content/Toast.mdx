```jsx live
const DemoComponent = () => {
  const toaster = useToast();

  const handleAddToastClick = () => {
    toaster.addToast({
      message: 'Details successfully saved!',
      state: 'success'
    })
  }

  return (
    <>
      <Button onClick={handleAddToastClick}>Add a toast</Button>
      <ToastContainer {...toaster/>
    </>
  )
}
```
